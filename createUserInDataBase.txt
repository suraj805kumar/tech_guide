CREATE USER ONLINE_CHAT IDENTIFIED BY CHAT;

GRANT CREATE SESSION to ONLINE_CHAT;
GRANT CREATE SYNONYM TO ONLINE_CHAT;
GRANT CREATE PROCEDURE TO ONLINE_CHAT;
GRANT CREATE TABLE TO ONLINE_CHAT;
GRANT CREATE VIEW TO ONLINE_CHAT;
grant all privileges to ONLINE_CHAT;
grant DBA to ONLINE_CHAT;

create table LoginUser(
LOGINID number(10) primary key,
USERNAME varchar2(100),
PASSWORD varchar2(100),
ONLINE_STATUS varchar2(10)
);

create table Friends(
LOGINID number(10) ,
FRIENDID number(10) primary key,
FRIENDS number(10),
CONSTRAINT LoginUser_FK FOREIGN KEY (LOGINID)
    REFERENCES LoginUser(LOGINID)
);
